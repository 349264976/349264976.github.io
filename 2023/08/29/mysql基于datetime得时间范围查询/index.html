mysql 基于时间datetime得查询

#查询当日得数据
select * from admins where TO_DAYS(createtime) = TO_DAYS(NOW())
#查询昨天的数据
SELECT * FROM admins WHERE TO_DAYS(NOW()) - TO_DAYS(createtime) = 1
#查询近7天的数据（包含今天）
SELECT * FROM admins WHERE DATE_SUB(CURDATE(), INTERVAL 7 DAY) <= createtime

#查询本周的数据
SELECT * FROM admins WHERE YEARWEEK(createtime,’%Y-%m-%d’) = YEARWEEK(NOW());
#查询上周的数据
SELECT * FROM admins WHERE YEARWEEK(createtime,’%Y-%m-%d’) = YEARWEEK(NOW())-1;
#查询近30天的数据（包含今天）
SELECT * FROM admins where DATE_SUB(CURDATE(), INTERVAL 30 DAY) <= createtime

#查询本月的数据
– SELECT * FROM table_name WHERE FROM_UNIXTIME(createtime, ‘%Y%m’) = DATE_FORMAT(CURDATE(), ‘%Y%m’)
SELECT * FROM admins WHERE YEAR(createtime) = YEAR(CURDATE()) AND MONTH(createtime) = MONTH(CURDATE());

#查询上月的数据
– SELECT * FROM admins WHERE PERIOD_DIFF(DATE_FORMAT(NOW(),’%Y%m’), FROM_UNIXTIME(createtime,’%Y%m’)) = 1
SELECT * FROM admins WHERE PERIOD_DIFF(DATE_FORMAT(NOW(), ‘%Y%m’), DATE_FORMAT(createtime, ‘%Y%m’)) = 1;

#查询本季度的数据
– SELECT * FROM table_name WHERE QUARTER(FROM_UNIXTIME(create_time))=QUARTER(NOW()) AND YEAR(FROM_UNIXTIME(create_time))=YEAR(NOW())
SELECT * FROM admins WHERE QUARTER(createtime) = QUARTER(NOW()) AND YEAR(createtime) = YEAR(NOW());

#查询上季度的数据
– SELECT * FROM table_name WHERE QUARTER(FROM_UNIXTIME(create_time)) = QUARTER(DATE_SUB(NOW(),INTERVAL 1 QUARTER)) AND YEAR(FROM_UNIXTIME(create_time))=YEAR(NOW())
SELECT * FROM admins WHERE QUARTER(createtime) = QUARTER(DATE_SUB(NOW(), INTERVAL 1 QUARTER)) AND YEAR(createtime) = YEAR(NOW());

#查询今年的数据
– SELECT * FROM table_name WHERE YEAR(FROM_UNIXTIME(create_time))=YEAR(NOW())
SELECT * FROM admins WHERE YEAR(createtime) = YEAR(NOW());

#查询去年的数据
– SELECT * FROM table_name WHERE YEAR(FROM_UNIXTIME(create_time)) = YEAR(DATE_SUB(NOW(),INTERVAL 1 YEAR))

SELECT * FROM admins WHERE YEAR(createtime) = YEAR(DATE_SUB(NOW(), INTERVAL 1 YEAR));
